Module CTrans_utils.Loops

Module that provides utility functions for translating different types of loops.

type loop_kind =
  1. | For of {
    1. init : ATDGenerated.Clang_ast_t.stmt;
    2. decl_stmt : ATDGenerated.Clang_ast_t.stmt;
    3. condition : ATDGenerated.Clang_ast_t.stmt;
    4. increment : ATDGenerated.Clang_ast_t.stmt;
    5. body : ATDGenerated.Clang_ast_t.stmt;
    }
  2. | While of {
    1. decl_stmt : ATDGenerated.Clang_ast_t.stmt option;
    2. condition : ATDGenerated.Clang_ast_t.stmt;
    3. body : ATDGenerated.Clang_ast_t.stmt;
    }
  3. | DoWhile of {
    1. condition : ATDGenerated.Clang_ast_t.stmt;
    2. body : ATDGenerated.Clang_ast_t.stmt;
    }
val get_cond : loop_kind -> ATDGenerated.Clang_ast_t.stmt
val get_body : loop_kind -> ATDGenerated.Clang_ast_t.stmt
