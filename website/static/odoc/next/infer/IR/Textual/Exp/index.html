Module Textual.Exp

type t =
| Var of Ident.t(*

pure variable: it is not an lvalue

*)
| Lvar of VarName.t(*

the address of a program variable

*)
| Field of {
exp : t;
tname : TypeName.t;
fname : FieldBaseName.t;
}		(*

field offset, fname must be declared in type tname

*)
| Index of t * t(*

an array index offset: exp1[exp2]

*)
| Const of Const.t
| Call of {
proc : Procname.qualified_name;
args : t list;
}
| Cast of Typ.t * t
val not : t -> t
