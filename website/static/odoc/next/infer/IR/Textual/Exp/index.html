Module Textual.Exp

type t =
| Var of Ident.t(*

pure variable: it is not an lvalue

*)
| Lvar of VarName.t(*

the address of a program variable

*)
| Lfield of {
exp : t;
tname : TypeName.t;
fname : FieldBaseName.t;
}		(*

field offset, fname must be declared in type tname

*)
| Lindex of t * t(*

an array index offset: exp1[exp2]

*)
| EConst of Const.t
| ECall of ProcBaseName.t * t list
| ECast of Typ.t * t
